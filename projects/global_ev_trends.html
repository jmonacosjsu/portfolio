# Top 10 countries by EV sales growth (20102023)
import matplotlib.pyplot as plt
top_growth = (
    df[(df['parameter'] == 'EV sales') & ~df['region'].isin(['World', 'Rest of the world', 'Europe', 'EU27'])]
    .groupby(['region', 'year'])['value'].sum()
    .unstack(fill_value=0)
    .assign(sales_growth=lambda x: x[2023] - x[2010])
    .sort_values('sales_growth', ascending=False)
    .head(10)
)

top_growth['sales_growth'].plot(kind='bar', figsize=(10, 6), color='skyblue', title='Top 10 Countries by EV Sales Growth (20102023)')
plt.ylabel('EV Sales Growth (Vehicles)')
plt.xlabel('Country')
plt.xticks(rotation=0)
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.tight_layout()
plt.show()

import matplotlib.pyplot as plt
import matplotlib.ticker as mtick

# Filter for EV sales between 2010 and 2023
filtered = df[
    (df['parameter'] == 'EV sales') &
    (df['year'].between(2010, 2023))
]

# Optionally adjust for units (comment out if already in vehicle count)
# filtered['value'] = filtered['value'] * 1000  # Uncomment if units are in thousands

# Group and pivot
grouped = (
    filtered.groupby(['mode', 'year'])['value']
    .sum()
    .unstack(fill_value=0)
    .T
)

# Optional: sort columns by total sales
grouped = grouped[grouped.sum().sort_values(ascending=False).index]

# Plot
ax = grouped.plot(figsize=(12, 6), linewidth=2, title='EV Sales by Vehicle Type (20102023)')
ax.set_ylabel('Sales (Vehicles)')
ax.set_xlabel('Year')
ax.yaxis.set_major_formatter(mtick.FuncFormatter(lambda x, _: f'{x/1e6:.1f}M'))  # Millions
plt.grid(axis='y', linestyle='--', alpha=0.6)
plt.legend(title='Vehicle Type')
plt.tight_layout()
plt.show()

# Question 3: How does EV stock share compare across countries?
# Filter stock share entries in 2023
stock_share = df[
    (df['parameter'] == 'EV stock share') &
    (df['year'] == 2023) &
    (~df['region'].isin(['World', 'Rest of the world', 'Europe', 'EU27']))
]

# Get max stock share per country
top_stock_share = (
    stock_share.groupby('region')['value']
    .max()  
    .sort_values(ascending=False)
    .head(20)
)

# Plot
top_stock_share.plot(kind='barh', figsize=(10, 6), color='mediumseagreen',
                     title='Top 20 Countries by EV Stock Share (2023)')
plt.xlabel('EV Stock Share (%)')
plt.ylabel('Country')
plt.gca().invert_yaxis()  
plt.grid(axis='x', linestyle='--', alpha=0.6)
plt.tight_layout()
plt.show()
