import seaborn as sns
import matplotlib.pyplot as plt

date_col = "order_date"  # or reservation_date
df_all[date_col] = pd.to_datetime(df_all[date_col], errors="coerce")

df_all["year"] = df_all[date_col].dt.year
df_all["month"] = df_all[date_col].dt.month_name()

# Group counts
monthly_counts = (
    df_all.groupby(["year","month"])
          .size()
          .reset_index(name="bookings")
)

# Pivot so months are columns
pivot = monthly_counts.pivot(index="year", columns="month", values="bookings")

# Reorder months JanDec
pivot = pivot[[ 'January','February','March','April','May','June',
                'July','August','September','October','November','December' ]]

plt.figure(figsize=(14,6))
sns.heatmap(pivot, annot=True, fmt="g", cmap="Greens")
plt.title("Monthly Bookings by Year")
plt.ylabel("Year")
plt.xlabel("Month")
